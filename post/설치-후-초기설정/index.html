# zsh-syntax-highlighting
(https://github.com/zsh-users/zsh-syntax-highlighting/blob/master/INSTALL.md)
[1] git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting

[2] vim ~/.zshrc

[3] plugins=(
      git
      zsh-syntax-highlighting
    )

[4] source ~/.zshrc


# zsh-autosuggestions

[1] curl -L https://github.com/pyenv/pyenv-installer/raw/master/bin/pyenv-installer | bash

[2] vim ~/.zshrc

[3] 아래 내용 넣기
export PATH="$HOME/.pyenv/bin:$PATH"
eval "$(pyenv init -)"
eval "$(pyenv virtualenv-init -)"

[4] 터미널 재시작 또는 source ~/.zshrc

# 파이썬 설치 전 필요 패키지
[5] 
sudo apt-get install -y make build-essential libssl-dev zlib1g-dev libbz2-dev \
libreadline-dev libsqlite3-dev wget curl llvm libncurses5-dev libncursesw5-dev \
xz-utils tk-dev libffi-dev liblzma-dev python-openssl

# 너무 최신 버전은 사용하지 말자
[6] pyenv install --list 를 입력하여 원하는 버전을 설치하자

[7] pyenv install 3.7.2

[8] pyenv versions 
    위 내용을 입력했을 때 아래와 같이 나온다면 [10] 내용을 입력하자
    * system (set by /home/gs/.pyenv/version)
    3.7.2

[9] python --version
     2.x.x

[10] pyenv global 3.7.2

[11] 재시작

[12] pyenv versions
     system
     * 3.7.2 (set by /home/gs/.pyenv/version)

[13] pyenv --version
     파이썬도 같이 버전업이 된 것을 확인할 수 있다
     Python 3.7.2
